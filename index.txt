        let currentDeck = [];

        document.getElementById('card-element').onclick = function() {
            if (!this.classList.contains('discarding')) this.classList.toggle('is-flipped');
        };

        function renderDeckVisuals(mode) {
            const container = document.getElementById('deck-stack');
            
            if (mode === 'normal') {
                container.innerHTML = '';
                const visualTotal = Math.min(currentDeck.length, 6);
                for (let i = 0; i < visualTotal; i++) {
                    const card = document.createElement('div');
                    card.className = 'stacked-card card-back-pattern';
                    card.style.top = `${i * 3}px`;
                    card.style.left = `${i * 3}px`;
                    card.style.zIndex = 10 - i;
                    container.appendChild(card);
                }
            } else if (mode === 'messy') {
                container.innerHTML = '';
                for (let i = 0; i < 12; i++) {
                    const card = document.createElement('div');
                    card.className = 'stacked-card card-back-pattern';
                    const rx = Math.random() * 200 - 100;
                    const ry = Math.random() * 120 - 60;
                    const rr = Math.random() * 100 - 50;
                    card.style.transform = `translate(${rx}px, ${ry}px) rotate(${rr}deg)`;
                    card.style.opacity = "0.8";
                    container.appendChild(card);
                }
            } else if (mode === 'collecting') {
                // Seleciona as cartas que já estão na mesa e move para o centro
                const cards = container.querySelectorAll('.stacked-card');
                cards.forEach((card, i) => {
                    card.style.transform = `translate(0px, 0px) rotate(0deg)`;
                    card.style.opacity = "1";
                    card.style.top = `${i * 2}px`;
                    card.style.left = `${i * 2}px`;
                });
            }
        }

        function startNewGame() {
            if (currentDeck.length === 0) {
                renderDeckVisuals('collecting'); // Primeiro anima
                
                // Espera o tempo da transição CSS antes de resetar os dados
                setTimeout(() => {
                    resetDeckData();
                }, 600);
            } else {
                resetDeckData();
            }
        }

        function resetDeckData() {
            currentDeck = [...allCountries];
            shuffleDeck();
            document.getElementById('main-menu').classList.add('hidden');
            document.getElementById('deck-scenario').classList.remove('hidden');
            updateDeckStatus();
        }

        function shuffleDeck() {
            currentDeck.sort(() => Math.random() - 0.5);
        }

        function updateDeckStatus() {
            const count = currentDeck.length;
            const drawBtn = document.getElementById('draw-btn');
            document.getElementById('deck-status').innerText = `Cards left: ${count}`;
            
            if (count === 0) {
                renderDeckVisuals('messy');
                drawBtn.innerText = "SHUFFLE AGAIN";
                drawBtn.style.background = "var(--reset-color)";
                drawBtn.onclick = startNewGame;
            } else {
                renderDeckVisuals('normal');
                drawBtn.innerText = "PICK A CARD";
                drawBtn.style.background = "var(--accent-color)";
                drawBtn.onclick = drawCard;
            }
        }

        function drawCard() {
            if (currentDeck.length === 0) return;
            const selected = currentDeck.pop();
            const card = document.getElementById('card-element');
            card.classList.remove('is-flipped', 'discarding');
            
            document.getElementById('country-name').innerText = selected.name;
            document.getElementById('country-flag').src = `https://flagcdn.com/w80/${selected.code}.png`;
            const container = document.getElementById('tips-container');
            container.innerHTML = "";
            selected.tips.forEach((tip, idx) => {
                container.innerHTML += `<li><strong>${idx + 1}.</strong> ${tip}</li>`;
            });

            document.getElementById('deck-scenario').classList.add('hidden');
            document.getElementById('card-scene').style.display = 'block';
document.getElementById('mini-reset').style.display = 'none';
        }

        function backToDeck(event) {
            event.stopPropagation();
            document.getElementById('card-element').classList.add('discarding');
            setTimeout(() => {
                document.getElementById('card-scene').style.display = 'none';
                document.getElementById('deck-scenario').classList.remove('hidden');
                updateDeckStatus();
document.getElementById('mini-reset').style.display = 'block';
            }, 750); 
        }
